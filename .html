def criar_tabuleiro(l, c): # cria a matriz do tabuleiro
    tabuleiro = []
    for i in range(l):
        linha = []
        for j in range(c):
            linha.append(0)
        tabuleiro.append(linha)
    return tabuleiro

def imprimir_tabuleiro(tabuleirotoPrint, pessoa='eu', nome_jogador=''): # 'eu' = tabuleiro próprio

def tratamento_coordenadas(entrada, numcolunas, numlinhas):
#converte coordenadas em formato letranumero em formato X, Y
    if len(entrada) < 2:
        print("Entrada inválida! Use o formato coluna + linha (ex. A5)")
        return -1, -1
    colunaString = entrada[0].upper()
    linhaString = entrada[1:]
    if not colunaString.isalpha():
        print("Entrada inválida! Use o formato coluna + linha (ex. A5)")
        return -1, -1 # valor inválido
    if not linhaString.isdigit():
        print("Entrada inválida! Use o formato coluna + linha (ex. A5)")
        return -1, -1
    coluna = ord(colunaString) - ord('A')
    linha = int(linhaString) - 1
    if not (0 <= linha < numlinhas and 0 <= coluna < numcolunas):
        print("Coordenadas fora dos limites do tabuleiro!!")
        return -1, -1
    return linha, coluna

def validar_posicao_navio(tabuleiroreal, tamanhoNavio, linhainicial, colunainicial, orientacao):  
# Verifica se a posição é válida para o navio
    numlinhas = len(tabuleiroreal)
    numcolunas = len(tabuleiroreal[0])
    violacoes = 0
    if orientacao == 'H':#verifica se cabe horizontalmente
        if colunainicial + tamanhoNavio > numcolunas:
            violacoes += 1 # Não cabe
        else:
            for j in range(tamanhoNavio):
                if tabuleiroreal[linhainicial][colunainicial + j] != 0:  # verifica se não é em branco
                    violacoes += 1
    elif orientacao == 'V': #verifica se cabe verticalmente
        if linhainicial + tamanhoNavio > numlinhas:
            violacoes += 1
        else:
            for i in range(tamanhoNavio):
                if tabuleiroreal[linhainicial + i][colunainicial] != 0:
                    violacoes += 1 #Espaço já ocupado
    else:
        violacoes += 1 # Considera violação casa a entrada seja inválida
    return violacoes == 0

def organizacao_manual(tabuleiroreal, embarcacoes_list, nome_jogador=""):

    numcolunas = len(tabuleiroreal[0])
    print(f"\n------ {nome_jogador}: POSICIONE SUAS EMBARCAÇÕES!!! ------")
    for nomeNavio, tamanhoNavio in embarcacoes_list:

        while posicionadostatus == 0:
            imprimir_tabuleiro(tabuleiroreal, pessoa = 'eu', nome_jogador=nome_jogador)
            print(f"\nPosicionando: {nomeNavio} (Tamanho: {tamanhoNavio})")
            entrada = input(f"Digite a coordenada inicial: ").upper()
            linhainicial, colunainicial = tratamento_coordenadas(entrada, numcolunas, numlinhas)
            if linhainicial == -1:
                continue
            entradaorientacao = input("Digite a orientação, horizontal ou vertical (H ou V):").upper()
            if entradaorientacao not in direcoes:
                print("Orientação inválida. Por favor, digite H ou V.")
                continue
            if validar_posicao_navio(tabuleiroreal, tamanhoNavio, linhainicial, colunainicial, entradaorientacao):
                if entradaorientacao == 'H':
                    for j in range(tamanhoNavio):
                        tabuleiroreal[linhainicial][colunainicial + j] = 1
                elif entradaorientacao == 'V':
                    for i in range(tamanhoNavio):
                        tabuleiroreal[linhainicial + i][colunainicial] = 1
                posicionadostatus = 1
                print(f"'{nomeNavio}' posicionado com sucesso!!")
                time.sleep(1.5)
            else:
                print(
                    "Não foi possível posicionar a embarcação aqui.")
                time.sleep(1.5)
    print(f"\nTodas as embarcações de {nome_jogador} foram posicionadas!!")
    imprimir_tabuleiro(tabuleiroreal, pessoa = 'eu', nome_jogador=nome_jogador)
    time.sleep(2)

def organizacao_aleatoria(tabuleiroreal, embarcacoes_list, nome_jogador=""):
    num_linhas = len(tabuleiroreal)
    num_colunas = len(tabuleiroreal[0])
    print(f"\nPosicionado as embarcações de {nome_jogador} aleatoriamente")
    for nomeNavio, tamanhoNavio in embarcacoes_list:
        posicionado = 0
        while posicionado == 0:
            linhainicial = random.randint(0, num_linhas - 1)
            colunainicial = random.randint(0, num_colunas - 1)
            orientacao = random.choice(direcoes)
            if validar_posicao_navio(tabuleiroreal, tamanhoNavio, linhainicial, colunainicial, orientacao):
                if orientacao == 'H':
                    for j in range(tamanhoNavio):
                        tabuleiroreal[linhainicial][colunainicial + j] = 1
                elif orientacao == 'V':
                    for i in range(tamanhoNavio):
                        tabuleiroreal[linhainicial + i][colunainicial] = 1
                posicionado += 1
    print(f"Embarcações de {nome_jogador} posicionadas.")
    time.sleep(1)

def mecanica_jogada(tabuleiroreal, tabuleirodisplay, linha, coluna):
    # tabela resultados:
    #0: água atingida
    #1: acerto novo
    #2: navio já atingido
    #3: água já atingida
    valueatual = tabuleiroreal[linha][coluna]
    valuedisplay = tabuleirodisplay[linha][coluna]
    if valuedisplay == 2:
        return "Você já atingiu esta parte de um navio!", 2
    elif valuedisplay == 3:
        return "Você já atingiu água aqui!!", 3
    if valueatual == 1:
        tabuleiroreal[linha][coluna] = 2
        tabuleirodisplay[linha][coluna] = 2
        return "BOMBA!", 1
    else:
        tabuleiroreal[linha][coluna] = 3
        tabuleirodisplay[linha][coluna] = 3
        return "ÁGUA!", 0

def navio_restante(tabuleiroreal):
    partes_restantes = 0
    for linha in tabuleiroreal:
        for valorcaixa in linha:
            if valorcaixa == 1:
                partes_restantes += 1
    return partes_restantes

            board_display[i][j] = '.'

    if nome_jogador == "COMPUTADOR":
        organizacao_aleatoria(board_real, embarcacoes, nome_jogador)
    else:

def limpar_terminal():
    os.system('cls' if os.name == 'nt' else 'clear')

def centralizar_texto(texto):
    largura_terminal = shutil.get_terminal_size().columns
    linhas = texto.split('\n')
    texto_centralizado = []
    for linha in linhas:
        espacos = max((largura_terminal - len(linha)) // 2, 0)
        texto_centralizado.append(' ' * espacos + linha)
    return '\n'.join(texto_centralizado)

                limpar_terminal()

                time.sleep(3)

        except KeyboardInterrupt:
            print("\nPrograma interrompido pelo usuário.")
            sys.exit()
